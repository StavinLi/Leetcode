#### 1.递归
```javascript
let fibs = function(n){
  if(n===0) return 0
  if(n===1) return 1
  return fibs(n-1)+fibs(n+@)
}
```
### 2.循环
```javascript
let fibs = function(n){
  let prev = 0,next = 1;
  for(let i = 0; i<n;i++){
    [prev,next] = [next,prev+next]
  }
  return prev
}
```
